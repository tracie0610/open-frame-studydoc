nHibernate的ORM框架映射的第二步
2008-03-15 10:37

实现右键生成JavaBean功能：

这个东西昨天实现的，基本原理也是比较简单。

这个时候在工程的里面有没有生成JavaBean 文件。

在表里面点击右键，选中生成JavaBean。即可在工程中生成JavaBean。现在没有做

保存对话框。比较懒！！！

运行完成后在工程的db包里面生成了TbUser.java文件。可以完全比较Hibernate生成的东东。

现在就把原理类与大家公开把！

/*
* To change this template, choose Tools | Templates
* and open the template in the editor.
*/

package org.jdbc.ui.db.io;

import java.io.BufferedWriter;
import java.io.File;
import java.io.FileWriter;
import java.io.IOException;
import java.util.List;
import org.jdbc.ui.db.ColumnEntity;

/**
*
* @author Banseon
*/
public class CreateBeans implements java.io.Serializable{
    

     /*该方法主要提供生成JavaBean的方法*/
    public void outPutBean(List<ColumnEntity> columns,String savePath,String fileName) throws IOException{
        File file = new File(savePath+"/"+fileName);
        FileWriter writer = new FileWriter(file);
        BufferedWriter out = new BufferedWriter(writer);
        String _path = savePath.replaceAll("/", ".");
        _path = _path.replaceAll("src.", "");
        String _fileName = fileName.replaceAll(".java", "");
        out.write("package "+_path+"; \n");
        out.write("/** \n");
        out.write("*create bean fo nHibernate project \n");
        out.write("*@antuthor banseon \n");
        out.write("*/ \n");
        out.write("public class "+_fileName +" implements java.io.Serializable{ \n");
        out.write("\n");
        for (ColumnEntity column :columns) {   
              out.write("\t private "+ this.typeToString(column.getColumnType()) + " " +column.getColumnName() +"; \n");
              out.write("\n");
        }
       
        for (ColumnEntity column :columns) {
           out.write("\t public void set"+this.nameToUp(column.getColumnName())+ " ("+this.typeToString(column.getColumnType())
                   +" " + column.getColumnName() + "){ \n");
           out.write("\t\t this."+column.getColumnName() + " = "+ column.getColumnName() +";\n");
           out.write("\t} \n");
           out.write("\n");
           out.write("\t public "+this.typeToString(column.getColumnType()) + " get"+this.nameToUp(column.getColumnName())+"(){ \n");
           out.write("\t\t return this."+column.getColumnName() +"; \n");
           out.write("\t} \n");
           out.write("\n");
        }
       
        out.write("} \n");
        out.flush();
        out.close();
    }
   
    /*对输入的属性名进行首字母大写的转换*/
    private String nameToUp(String name){
        String _temp1 = name.substring(0, 1) ;
        _temp1 = _temp1.toUpperCase();
        String _temp2 = name.substring(1);
        return _temp1+_temp2;
    }
    /*将数据库中字段的类型转换成Java中所需要的类型*/
    private String typeToString(int i) {
        String ret = "";
        switch (i) {
            case (1):
                //ret = "CHAR";
                ret = "char";
                break;
            case (2):
                //ret = "NUMERIC";
                ret = "int";
                break;
            case (3):
                //ret = "DECIMAL";
                ret = "String";
                break;
            case (4):
                //ret = "INTEGER";
                ret = "Integer";
                break;
            case (5):
                //ret = "SMALLINT";
                ret = "String";
                break;
            case (6):
                //ret = "FLOAT";
                ret = "float";
                break;
            case (8):
                //ret = "DOUBLE";
                ret = "double";
                break;
            case (12):
                //ret = "VARCHAR";
                ret = "String";
                break;
            case (91):
                //ret = "DATE";
                ret = "String";
                break;
            default:
                //ret = "other";
                ret = "String";
        }
        return ret;
    }
   
    public static void main(String[] args) {
        CreateBeans bean = new CreateBeans();
        System.out.println(bean.nameToUp("Banseon"));
        String temp = "org/ejb/db";
        System.out.println(temp.replaceAll("/", "."));
    }


}

/***********************************************************************************************/

在右键功能中直接实现生成JavaBean即可。
